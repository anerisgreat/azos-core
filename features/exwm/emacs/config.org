** Smart buffer naming

#+begin_src emacs-lisp
(defvar azos/exwm/title-debounce-timers (make-hash-table :test 'eq)
  "Map of buffer → pending title-rename timer, used to debounce qutebrowser windows.")

(defun azos/exwm/debounce-title-rename (title buf)
  "Schedule a rename of BUF to TITLE after 0.5s, cancelling any pending rename for BUF."
  (let ((existing (gethash buf azos/exwm/title-debounce-timers)))
    (when existing (cancel-timer existing)))
  (puthash buf
           (run-with-timer 0.5 nil
             (lambda ()
               (remhash buf azos/exwm/title-debounce-timers)
               (when (buffer-live-p buf)
                 (with-current-buffer buf
                   (exwm-workspace-rename-buffer title)))))
           azos/exwm/title-debounce-timers))

** Dedicated processes

We'd want to be able to quickly map processes (Spotify, etc.) to
keybindings, and have dedicated buffers for them (so they don't reopen).

We first define variables to be used later, and a function that checks, for each
new process, should it be displayed in a new buffer/tab.

#+begin_src emacs-lisp
(defvar azos/exwm/startproc-regex-buffname-list nil
  "Match between buffer and tab.")
(defun azos/exwm/startproc-check-for-buff-entry ()
  "On new buffer, check if requested to display differently"
  (let ((entry (cdr (car (seq-filter
                     (lambda (e) (string-match (car e) (buffer-name)))
                    azos/exwm/startproc-regex-buffname-list
                     )))))
    (if entry
        (let ((rn (nth 0 entry)) (tn (nth 1 entry)) (buff (current-buffer)))
            ;; Too much complication for renaming
            ;; (if requested-name (rename-buffer requested-name))
            (progn
                (if tn (tab-bar-switch-to-tab tn))
                (switch-to-buffer buff)
                )))))

(add-hook 'exwm-manage-finish-hook 'azos/exwm/startproc-check-for-buff-entry)
#+end_src

#+begin_src emacs-lisp
(defun azos/exwm/start-proc-dedicated
    (name tab-name procregex proc &rest args)
    (let ((buff (car (seq-filter (lambda (b
)
            (string-match procregex (buffer-name b)))
                    (buffer-list)))))
        (if buff
            ;; Found buffer, display
            (progn
                (if tab-name (tab-bar-switch-to-tab tab-name))
                (switch-to-buffer buff)
                )
            ;;No known buffer, add entry to alist
            (let ((new-entry `(,procregex ,tab-name ,name))) (progn
                (add-to-list 'azos/exwm/startproc-regex-buffname-list
                             new-entry)
                (apply 'start-process name nil proc
                        args))))))

(defvar azos/audio/status-string nil "Holds audio string")

(defun azos/audio/parse-status-output (output)
  "Parse combined wpctl output (get-volume then inspect) into a display string."
  (let* ((lines (split-string (string-trim output) "\n"))
         (vol-line (or (car lines) ""))
         (sink-name (or (cadr lines) ""))
         ;; Parse volume: "Volume: 0.45 [MUTED]" → integer percentage
         (vol-parts (split-string vol-line "\\."))
         (dec (or (cadr vol-parts) "0"))
         (volume (string-to-number (car (split-string dec "[^0-9]"))))
         (ismute (string-match-p "\\[MUTED\\]" vol-line))
         ;; Pick icon from sink name
         (icon (cond
                ((string-match-p "headset\\|headphone" sink-name) "🎧")
                ((string-match-p "bluez" sink-name) "🎧")
                ((string-match-p "speaker" sink-name) "🔊")
                ((string-match-p "analog" sink-name) "🔊")
                (t "🔈"))))
    (concat icon (if ismute "  🔇" (format "%3d" volume)))))

(defun azos/audio/set-status-string-sentinel (process event)
  "Async sentinel: update `azos/audio/status-string' from process output."
  (when (memq (process-status process) '(exit signal))
    (let ((output (with-current-buffer (process-buffer process)
                    (buffer-string))))
      (setq azos/audio/status-string
            (condition-case nil
                (azos/audio/parse-status-output output)
              (error nil)))
      (kill-buffer (process-buffer process)))))

#+begin_src emacs-lisp
(defun azos/storage--lsblk-field (dev field)
  "Get FIELD from lsblk DEV hash table."
  (when (hash-table-p dev)
    (gethash field dev)))

(defun azos/storage--list-devices ()
  "Return flat list of partitions/devices from lsblk, with bare disks included."
  (let* ((json-str (shell-command-to-string
                    "lsblk --json -o NAME,SIZE,LABEL,MOUNTPOINT,TYPE"))
         (data (condition-case nil (json-parse-string json-str) (error nil)))
         (devs '()))
    (when (hash-table-p data)
      (seq-do
       (lambda (dev)
         (let ((children (azos/storage--lsblk-field dev "children")))
           (if (and (vectorp children) (> (length children) 0))
               (seq-do (lambda (child) (push child devs)) children)
             (push dev devs))))
       (gethash "blockdevices" data)))
    (nreverse devs)))

(defun azos/storage--null-p (val)
  "Return t if VAL is a JSON null (i.e. :null)."
  (eq val :null))

(defun azos/storage--label-str (dev)
  "Format label for DEV, returning empty string if absent."
  (let ((label (azos/storage--lsblk-field dev "label")))
    (if (and label (not (azos/storage--null-p label)))
        (format "  [%s]" label)
      "")))

(defun azos/storage/mount ()
  "Mount an unmounted block device using udisksctl."
  (interactive)
  (let* ((devs (azos/storage--list-devices))
         (unmounted (seq-filter
                     (lambda (d)
                       (let ((mp (azos/storage--lsblk-field d "mountpoint"))
                             (type (azos/storage--lsblk-field d "type"))
                             (size (azos/storage--lsblk-field d "size")))
                         (and (member type '("part" "disk"))
                              (not (equal size "0B"))
                              (or (null mp) (azos/storage--null-p mp)))))
                     devs))
         (choices (mapcar
                   (lambda (d)
                     (let ((name (azos/storage--lsblk-field d "name"))
                           (size (azos/storage--lsblk-field d "size")))
                       (cons (format "/dev/%s  %s%s" name size (azos/storage--label-str d))
                             (format "/dev/%s" name))))
                   unmounted)))
    (if choices
        (let* ((selected (completing-read "Mount device: " (mapcar #'car choices) nil t))
               (device (cdr (assoc selected choices))))
          (message "%s" (string-trim
                         (shell-command-to-string
                          (format "udisksctl mount -b %s" device)))))
      (message "No unmounted partitions found."))))

(defun azos/storage/unmount ()
  "Unmount a mounted block device using udisksctl."
  (interactive)
  (let* ((devs (azos/storage--list-devices))
         (mounted (seq-filter
                   (lambda (d)
                     (let ((mp (azos/storage--lsblk-field d "mountpoint")))
                       (and mp
                            (not (azos/storage--null-p mp))
                            (not (string-prefix-p "/nix" mp))
                            (not (string-prefix-p "/boot" mp))
                            (not (equal mp "/"))
                            (not (equal mp "[SWAP]")))))
                   devs))
         (choices (mapcar
                   (lambda (d)
                     (let ((name (azos/storage--lsblk-field d "name"))
                           (size (azos/storage--lsblk-field d "size"))
                           (mp   (azos/storage--lsblk-field d "mountpoint")))
                       (cons (format "/dev/%s  %s  -> %s%s" name size mp (azos/storage--label-str d))
                             (format "/dev/%s" name))))
                   mounted)))
    (if choices
        (let* ((selected (completing-read "Unmount device: " (mapcar #'car choices) nil t))
               (device (cdr (assoc selected choices))))
          (message "%s" (string-trim
                         (shell-command-to-string
                          (format "udisksctl unmount -b %s" device)))))
      (message "No mounted user partitions found."))))

* Screen Capture

Functions for recording the screen to video (mp4) or GIF using =ffmpeg=.

- =c r= — record full screen to mp4
- =c w= — record current Emacs window to mp4
- =c g= — record full screen, stop produces a GIF
- =c f= — record current Emacs window, stop produces a GIF
- =c s= — stop recording (and convert to GIF if applicable)

#+begin_src emacs-lisp
(defvar azos/screencast/process nil "Active screen recording process.")
(defvar azos/screencast/pending-gif nil "GIF output path for post-stop conversion, or nil.")
(defvar azos/screencast/temp-file nil "Temp mp4 file used when recording to GIF.")

(defun azos/screencast--display ()
  "Return the current X display string."
  (or (getenv "DISPLAY") ":0"))

(defun azos/screencast--decode-pos (val)
  "Decode frame position VAL, which may be an integer or a (+ N) list."
  (if (consp val) (cadr val) (or val 0)))

(defun azos/screencast--screen-geometry ()
  "Return (x y w h) for the full screen."
  (let* ((out (shell-command-to-string "xdpyinfo | grep -m1 dimensions"))
         (w (if (string-match "\\([0-9]+\\)x\\([0-9]+\\)" out) (match-string 1 out) "1920"))
         (h (if (string-match "\\([0-9]+\\)x\\([0-9]+\\)" out) (match-string 2 out) "1080")))
    (list 0 0 w h)))

(defun azos/screencast--window-geometry ()
  "Return (x y w h) for the current Emacs window, in screen coordinates."
  (let* ((edges (window-pixel-edges))
         (fl (azos/screencast--decode-pos (frame-parameter nil 'left)))
         (ft (azos/screencast--decode-pos (frame-parameter nil 'top)))
         (x  (+ fl (nth 0 edges)))
         (y  (+ ft (nth 1 edges)))
         (w  (- (nth 2 edges) (nth 0 edges)))
         (h  (- (nth 3 edges) (nth 1 edges))))
    (list x y w h)))

(defun azos/screencast--start (x y w h output-file)
  "Start an ffmpeg x11grab recording of region (X,Y,WxH) to OUTPUT-FILE."
  (when (process-live-p azos/screencast/process)
    (user-error "A recording is already in progress; call azos/screencast/stop first"))
  (setq azos/screencast/process
        (start-process "azos-screencast" nil
                       "ffmpeg" "-y"
                       "-f" "x11grab"
                       "-r" "30"
                       "-s" (format "%sx%s" w h)
                       "-i" (format "%s+%s,%s" (azos/screencast--display) x y)
                       "-c:v" "libx264" "-preset" "ultrafast"
                       output-file))
  (message "Recording started → %s" output-file))

(defun azos/screencast/stop ()
  "Stop the current recording, converting to GIF if one was requested."
  (interactive)
  (if (not (process-live-p azos/screencast/process))
      (message "No recording in progress.")
    (let ((proc azos/screencast/process))
      (setq azos/screencast/process nil)
      (set-process-sentinel
       proc
       (lambda (_proc _event)
         (let ((gif azos/screencast/pending-gif)
               (tmp azos/screencast/temp-file))
           (setq azos/screencast/pending-gif nil
                 azos/screencast/temp-file nil)
           (if gif
               (progn
                 (message "Converting to GIF…")
                 (set-process-sentinel
                  (start-process
                   "azos-gif" nil "ffmpeg" "-i" tmp
                   "-vf" (concat "fps=15,scale=1280:-1:flags=lanczos,"
                                 "split[s0][s1];[s0]palettegen[p];[s1][p]paletteuse")
                   "-loop" "0" gif)
                  (lambda (gif-proc _event)
                    (when (memq (process-status gif-proc) '(exit signal))
                      (delete-file tmp)
                      (message "GIF saved → %s" gif)))))
             (message "Recording stopped.")))))
      (signal-process proc 'SIGINT))))
